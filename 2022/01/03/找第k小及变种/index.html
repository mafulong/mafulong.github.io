class Solution:
    def nthSuperUglyNumber(self, n: int, primes: List[int]) -> int:
        import heapq
        hpush, hpop = heapq.heappush,heapq.heappop
        h = []
        for i in range(len(primes)):
            hpush(h, (primes[i], i, 0))
        ans = [1]
        for i in range(1,n):
            if not h:
                return -1
            val = h[0][0]
            ans.append(val)
            while h and h[0][0] == val:
                _,j,idx = hpop(h)
                # print(idx,ans)
                hpush(h, (ans[idx]*primes[j], j, idx+1))
        return ans[-1]
