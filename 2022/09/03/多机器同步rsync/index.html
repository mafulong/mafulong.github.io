多机器同步rsync

tb () {
	x=$(pwd)
	echo $x
	rsync -avz --delete --exclude='build' $x/ ${DEV_A}:/home/${x:6}/
}

get () {
	x=$(pwd)
	echo $x
	rsync -avz --delete --exclude='' ${DEV_A}:/home/${x:6}/ $x/
}

  • Linux: ionotify
