class Solution:
    def medianSlidingWindow(self, nums: List[int], k: int) -> List[float]:
        import sortedcontainers
        if k > len(nums):
            return []
        d = sortedcontainers.SortedList()
        def get_median():
            if k % 2 == 1:
                return d[len(d) // 2]
            else:
                return (d[(len(d) // 2) - 1] + d[len(d) // 2])/2
        for i in range(k - 1):
            d.add(nums[i])
        ans = []
        for i in range(k-1, len(nums)):
            d.add(nums[i])
            if i - k >= 0:
                d.discard(nums[i-k])
            # print(len(d))
            t = get_median()
            ans.append(t)
        return ans
