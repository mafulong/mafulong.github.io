class Solution:
    def minCost(self, nums: List[int], cost: List[int]) -> int:
        d = list(zip(nums, cost))
        d = sorted(d)
        mid = None
        count = 0
        total = sum(cost)
        for a, b in d:
            count += b
            if count >= total // 2:
                mid = a
                break
        return sum(abs(a - mid) * c for a, c in d)
