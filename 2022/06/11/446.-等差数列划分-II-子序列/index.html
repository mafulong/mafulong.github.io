class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        N = len(nums)
        # иЎЁзӨәд»Ҙiз»“е°ҫпјҢе…¬е·®дёәdзҡ„еӯҗеәҸеҲ—ж•°йҮҸпјҢиҝҷйҮҢеӯҗеәҸеҲ—й•ҝеәҰеҸҜиғҪдёә2. d = num[i] - num[j], жүҖд»Ҙdзҡ„ж•°йҮҸ<= N. 
        # е…ҲиҖғиҷ‘iз»“е°ҫпјҢз»“жһңiз»“е°ҫеҒҡдёҚеҲ°дёҚйҮҚдёҚжјҸпјҢеҸӘиғҪеҠ жӣҙеӨҡз»ҙеәҰжқҘи®Ўз®—гҖӮ
        dp = [collections.defaultdict(int) for _ in range(N)]
        ans = 0
        for i in range(N):
            for j in range(i):
                d = nums[i] - nums[j]
                dp[i][d] += dp[j][d] + 1
                if dp[j][d]:
                    ans += dp[j][d]
        return ans
