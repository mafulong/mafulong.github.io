class Solution:
    def isScramble(self, s1: str, s2: str) -> bool:
        if len(s1) != len(s2):
            return False
        if s1 == "" or s2 == "":
            return False
        n = len(s1)
        import collections
        import functools
        @functools.lru_cache(None)
        def dfs(i, j, l):
            # counter‰∏çÂêåÔºåfalse
            counter = collections.Counter()
            for k in range(i, i + l):
                counter[s1[k]] += 1
            for k in range(j, j + l):
                counter[s2[k]] -= 1
            for o in range(ord('a'), ord('z') + 1):
                if counter[chr(o)] != 0:
                    return False
            # a == b
            equal = True
            for k in range(l):
                if s1[i + k] != s2[j + k]:
                    equal = False
                    break
            if equal:
                return True

            # split it into two seq
            for l1 in range(1, l):
                if dfs(i, j, l1) and dfs(i + l1, j + l1, l - l1):
                    return True
                if dfs(i, j + l - l1, l1) and dfs(i + l1, j, l - l1):
                    return True
            return False

        r = dfs(0, 0, n)
        return r
