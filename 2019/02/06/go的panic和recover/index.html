goуџёpanicтњїrecover

2019/02/06 Go тЁ▒ 1653 тГЌ№╝їу║д 5 тѕєжњЪ

С╗ђС╣ѕТў» panic№╝Ъ

тюе Go У»ГУеђСИГ№╝їуеІт║ЈСИГСИђУѕгТў»Сй┐ућежћЎУ»»ТЮЦтцёуљєт╝ѓтИИТЃЁтєхсђѓт»╣С║јуеІт║ЈСИГтЄ║уј░уџётцДжЃетѕєт╝ѓтИИТЃЁтєх№╝їжћЎУ»»т░▒ти▓у╗ЈтцЪућеС║єсђѓ

СйєтюеТюЅС║ЏТЃЁтєх№╝їтйЊуеІт║ЈтЈЉућЪт╝ѓтИИТЌХ№╝їТЌаТ│Ћу╗Ду╗ГУ┐љУАїсђѓтюеУ┐ЎуДЇТЃЁтєхСИІ№╝їТѕЉС╗гС╝џСй┐уће panic ТЮЦу╗ѕТГбуеІт║ЈсђѓтйЊтЄйТЋ░тЈЉућЪ panic ТЌХ№╝їт«ЃС╝џу╗ѕТГбУ┐љУАї№╝їтюеТЅДУАїт«їТЅђТюЅуџёт╗ХУ┐ЪтЄйТЋ░тљј№╝їуеІт║ЈТјДтѕХУ┐ћтЏътѕ░У»ЦтЄйТЋ░уџёУ░ЃућеТќ╣сђѓУ┐ЎТаиуџёУ┐ЄуеІС╝џСИђуЏ┤ТїЂу╗ГСИІтј╗№╝їуЏ┤тѕ░тйЊтЅЇтЇЈуеІуџёТЅђТюЅтЄйТЋ░жЃйУ┐ћтЏъжђђтЄ║№╝їуёХтљјуеІт║ЈС╝џТЅЊтЇ░тЄ║ panic С┐АТЂ»№╝їТјЦуЮђТЅЊтЇ░тЄ║таєТаѕУиЪУИф№╝ѕStack Trace№╝Ѕ№╝їТюђтљјуеІт║Ју╗ѕТГбсђѓтюеу╝ќтєЎСИђСИфуц║СЙІуеІт║Јтљј№╝їТѕЉС╗гт░▒УЃйтЙѕтЦйтю░уљєУДБУ┐ЎСИфТдѓт┐хС║єсђѓ

тюеТюгТЋЎуеІжЄї№╝їТѕЉС╗гУ┐ўС╝џТјЦуЮђУ«еУ«║№╝їтйЊуеІт║ЈтЈЉућЪ panic ТЌХ№╝їСй┐уће recover тЈ»С╗ЦжЄЇТќ░УјитЙЌт»╣У»ЦуеІт║ЈуџёТјДтѕХсђѓ

тЈ»С╗ЦУ«цСИ║ panic тњї recover СИјтЁХС╗ќУ»ГУеђСИГуџё try-catch-finally У»ГтЈЦу▒╗С╝╝№╝їтЈфСИЇУ┐ЄСИђУѕгТѕЉС╗гтЙѕт░ЉСй┐уће panic тњї recoverсђѓУђїтйЊТѕЉС╗гСй┐ућеС║є panic тњї recover ТЌХ№╝їС╣ЪС╝џТ»ћ try-catch-finally ТЏ┤тіаС╝ўжЏЁ№╝їС╗БуаЂТЏ┤тіаТЋ┤Т┤Ђсђѓ

С╗ђС╣ѕТЌХтђЎт║ћУ»ЦСй┐уће panic№╝Ъ

жюђУдЂТ│еТёЈуџёТў»№╝їСйат║ћУ»Цт░йтЈ»УЃйтю░Сй┐ућежћЎУ»»№╝їУђїСИЇТў»Сй┐уће panic тњї recoverсђѓтЈфТюЅтйЊуеІт║ЈСИЇУЃйу╗Ду╗ГУ┐љУАїуџёТЌХтђЎ№╝їТЅЇт║ћУ»ЦСй┐уће panic тњї recover Тю║тѕХсђѓ

panic ТюЅСИцСИфтљѕуљєуџёућеСЙІсђѓ

тЈЉућЪС║єСИђСИфСИЇУЃйТЂбтцЇуџёжћЎУ»»№╝їТГцТЌХуеІт║ЈСИЇУЃйу╗Ду╗ГУ┐љУАїсђѓ СИђСИфСЙІтГљт░▒Тў» web ТюЇтіАтЎеТЌаТ│Ћу╗Љт«џТЅђУдЂТ▒ѓуџёуФ»тЈБсђѓтюеУ┐ЎуДЇТЃЁтєхСИІ№╝їт░▒т║ћУ»ЦСй┐уће panic№╝їтЏаСИ║тдѓТъюСИЇУЃйу╗Љт«џуФ»тЈБ№╝їтЋЦС╣ЪтЂџСИЇС║єсђѓ

тЈЉућЪС║єСИђСИфу╝ќуеІСИіуџёжћЎУ»»сђѓ тЂЄтдѓТѕЉС╗гТюЅСИђСИфТјЦТћХТїЄжњѕтЈѓТЋ░уџёТќ╣Т│Ћ№╝їУђїтЁХС╗ќС║║Сй┐уће nil СйюСИ║тЈѓТЋ░У░ЃућеС║єт«ЃсђѓтюеУ┐ЎуДЇТЃЁтєхСИІ№╝їТѕЉС╗гтЈ»С╗ЦСй┐уће panic№╝їтЏаСИ║У┐ЎТў»СИђСИфу╝ќуеІжћЎУ»»№╝џуће nil тЈѓТЋ░У░ЃућеС║єСИђСИфтЈфУЃйТјЦТћХтљѕТ│ЋТїЄжњѕуџёТќ╣Т│Ћсђѓ

тЈЉућЪ panic ТЌХуџё defer

ТѕЉС╗гжЄЇТќ░Тђ╗у╗ЊСИђСИІ panic тЂџС║єС╗ђС╣ѕсђѓтйЊтЄйТЋ░тЈЉућЪ panic ТЌХ№╝їт«ЃС╝џу╗ѕТГбУ┐љУАї№╝їтюеТЅДУАїт«їТЅђТюЅуџёт╗ХУ┐ЪтЄйТЋ░тљј№╝їуеІт║ЈТјДтѕХУ┐ћтЏътѕ░У»ЦтЄйТЋ░уџёУ░ЃућеТќ╣сђѓУ┐ЎТаиуџёУ┐ЄуеІС╝џСИђуЏ┤ТїЂу╗ГСИІтј╗№╝їуЏ┤тѕ░тйЊтЅЇтЇЈуеІуџёТЅђТюЅтЄйТЋ░жЃйУ┐ћтЏъжђђтЄ║№╝їуёХтљјуеІт║ЈС╝џТЅЊтЇ░тЄ║ panic С┐АТЂ»№╝їТјЦуЮђТЅЊтЇ░тЄ║таєТаѕУиЪУИф№╝їТюђтљјуеІт║Ју╗ѕТГбсђѓ

тюеСИіжЮбуџёСЙІтГљСИГ№╝їТѕЉС╗гТ▓АТюЅт╗ХУ┐ЪУ░ЃућеС╗╗СйЋтЄйТЋ░сђѓтдѓТъюТюЅт╗ХУ┐ЪтЄйТЋ░№╝їС╝џтЁѕУ░Ѓућет«Ѓ№╝їуёХтљјуеІт║ЈТјДтѕХУ┐ћтЏътѕ░тЄйТЋ░У░ЃућеТќ╣сђѓ

recover

recover Тў»СИђСИфтєЁт╗║тЄйТЋ░№╝їућеС║јжЄЇТќ░УјитЙЌ panic тЇЈуеІуџёТјДтѕХсђѓ

recover тЄйТЋ░уџёТаЄуГЙтдѓСИІТЅђуц║№╝џ

func recover() interface{}

тЈфТюЅтюет╗ХУ┐ЪтЄйТЋ░уџётєЁжЃе№╝їУ░Ѓуће recover ТЅЇТюЅућесђѓтюет╗ХУ┐ЪтЄйТЋ░тєЁУ░Ѓуће recover№╝їтЈ»С╗ЦтЈќтѕ░ panic уџёжћЎУ»»С┐АТЂ»№╝їт╣ХСИћтЂюТГб panic у╗ГтЈЉС║ІС╗Х№╝ѕPanicking Sequence№╝Ѕ№╝їуеІт║ЈУ┐љУАїТЂбтцЇТГБтИИсђѓтдѓТъютюет╗ХУ┐ЪтЄйТЋ░уџётцќжЃеУ░Ѓуће recover№╝їт░▒СИЇУЃйтЂюТГб panic у╗ГтЈЉС║ІС╗Хсђѓ

ТѕЉС╗гТЮЦС┐«Тћ╣СИђСИІуеІт║Ј№╝їтюетЈЉућЪ panic С╣Ітљј№╝їСй┐уће recover ТЮЦТЂбтцЇТГБтИИуџёУ┐љУАїсђѓ

package main

import (  
    "fmt"
)

func recoverName() {  
    if r := recover(); r!= nil {
        fmt.Println("recovered from ", r)
    }
}

func fullName(firstName *string, lastName *string) {  
    defer recoverName()
    if firstName == nil {
        panic("runtime error: first name cannot be nil")
    }
    if lastName == nil {
        panic("runtime error: last name cannot be nil")
    }
    fmt.Printf("%s %s\n", *firstName, *lastName)
    fmt.Println("returned normally from fullName")
}

func main() {  
    defer fmt.Println("deferred call in main")
    firstName := "Elon"
    fullName(&firstName, nil)
    fmt.Println("returned normally from main")
}
