void InOrderThreading(BiThrTree &Thrt,BiThrTree T)  
{  
    /*дёӯеәҸйҒҚеҺҶдәҢеҸүж ‘TпјҢе№¶е°Ҷе…¶дёӯеәҸзәҝзҙўеҢ–пјҢThrtжҢҮеҗ‘еӨҙз»“зӮ№*/  
    Thrt=new BiThrNode;         /*е»әеӨҙз»“еӨҙ*/  
    Thrt->LTag=0;               /*еӨҙз»“зӮ№жңүе·Ұеӯ©еӯҗпјҢиӢҘж ‘йқһз©әпјҢеҲҷе…¶е·Ұеӯ©еӯҗдёәж ‘ж №*/         
    Thrt->RTag=1;               /*еӨҙз»“зӮ№зҡ„еҸіеӯ©еӯҗжҢҮй’ҲдёәеҸізәҝзҙў*/  
    Thrt->rchild=Thrt;          /*еҲқе§ӢеҢ–ж—¶еҸіжҢҮй’ҲжҢҮеҗ‘иҮӘе·ұ*/  
    if(!T) Thrt->lchild=Thrt;   /*иӢҘж ‘дёәз©әпјҢеҲҷе·ҰжҢҮй’Ҳд№ҹжҢҮеҗ‘иҮӘе·ұ*/  
    else  
    {  
        Thrt->lchild=T; pre=Thrt; /*еӨҙз»“зӮ№зҡ„е·Ұеӯ©еӯҗжҢҮеҗ‘ж №пјҢpreеҲқеҖјжҢҮеҗ‘еӨҙз»“зӮ№*/  
        InThreading(T);          /*и°ғз”ЁдёҠиҝ°з®—жі•пјҢеҜ№д»ҘTдёәж №зҡ„дәҢеҸүж ‘иҝӣиЎҢдёӯеәҸзәҝзҙўеҢ–*/  
        pre->rchild=Thrt;        /*з®—жі•з»“жқҹеҗҺпјҢpreдёәжңҖеҸіз»“зӮ№пјҢpreзҡ„еҸізәҝзҙўжҢҮеҗ‘еӨҙз»“зӮ№*/  
        pre->RTag=1;  
        Thrt->rchild=pre;        /*еӨҙз»“зӮ№зҡ„еҸізәҝзҙўжҢҮеҗ‘pre*/  
    }   
}

1)е·Ұеӯҗж ‘зәҝзҙўеҢ–

в‘ еҰӮжһңpзҡ„е·Ұеӯ©еӯҗдёәз©әпјҢеҲҷз»ҷpеҠ дёҠе·ҰзәҝзҙўпјҢе°Ҷе…¶LTagзҪ®дёә1пјҢи®©pзҡ„е·Ұеӯ©еӯҗжҢҮй’ҲжҢҮеҗ‘pre(еүҚй©ұ);

в‘ЎеҰӮжһңpreзҡ„еҸіеӯ©еӯҗдёәз©әпјҢеҲҷз»ҷpreеҠ дёҠеҸізәҝзҙўпјҢе°Ҷе…¶RTagзҪ®дёә1пјҢи®©preзҡ„еҸіеӯ©еӯҗжҢҮй’ҲжҢҮеҗ‘p(еҗҺз»§);

3)е°ҶpreжҢҮеҗ‘еҲҡи®ҝй—®иҝҮзҡ„з»“зӮ№pпјҢеҚіpre = p;

4)еҸіеӯҗж ‘зәҝзҙўеҢ–гҖӮ

void InThreading(BiThrTree p)  
{  
    /*preжҳҜе…ЁеұҖеҸҳйҮҸпјҢеҲқе§ӢеҢ–ж—¶е…¶еҸіеӯ©еӯҗжҢҮй’Ҳдёәз©әпјҢдҫҝдәҺеңЁж ‘зҡ„жңҖе·ҰзӮ№ејҖе§Ӣе»әзәҝзҙў*/  
    if(p)  
    {  
        InThreading(p->lchild);            /*е·Ұеӯҗж ‘йҖ’еҪ’зәҝзҙўеҢ–*/  
        if(!p->lchild)                      /*pзҡ„е·Ұеӯ©еӯҗдёәз©ә*/  
        {                    
            p->LTag=1;                     /*з»ҷpеҠ дёҠе·Ұзәҝзҙў*/  
            p->lchild=pre;                 /*pзҡ„е·Ұеӯ©еӯҗжҢҮй’ҲжҢҮеҗ‘pre(еүҚй©ұ)*/  
        }  
        else  
        {  
            p->LTag=0;  
        }  
        if(!pre->rchild)                  /*preзҡ„еҸіеӯ©еӯҗдёәз©ә*/  
        {  
            pre->RTag=1;                  /*з»ҷpreеҠ дёҠеҸізәҝзҙў*/  
            pre->rchild=p;                /*preзҡ„еҸіеӯ©еӯҗжҢҮй’ҲжҢҮеҗ‘p(еҗҺз»§)*/  
        }  
        else  
        {  
            pre->RTag=0;  
        }  
        pre=p;                            /*дҝқжҢҒpreжҢҮеҗ‘pзҡ„еүҚй©ұ*/  
        InThreading(p->rchild);           /*еҸіеӯҗж ‘йҖ’еҪ’зәҝзҙўеҢ–*/  
    }  
}

Node * ThreadedBinaryTree::get_successor(Node * node)
{
    if (node->right_thread)
        return node->right;
    Node * p = node->right;
    while (p->left_thread == false)  //е·ІзәҝзҙўеҢ–пјҢж•…жӯӨеӨ„еҸӘиғҪз”Ёleft_threadжқҘеҲӨж–ӯе·Ұеӯҗж ‘зҡ„жғ…еҶө
        p = p->left;
    return p;
}

Node * ThreadedBinaryTree::get_precursor(Node * node)
{
    if (node->left_thread)
        return node->left;
    Node * p = node->left;
    while (p->right_thread == false)  //е·ІзәҝзҙўеҢ–пјҢж•…жӯӨеӨ„еҸӘиғҪз”Ёright_threadжқҘеҲӨж–ӯеҸіеӯҗж ‘зҡ„жғ…еҶө
        p = p->right;
    return p;
}
