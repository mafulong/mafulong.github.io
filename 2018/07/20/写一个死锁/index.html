еҶҷдёҖдёӘжӯ»й”ҒзЁӢеәҸ

еҫҲеӨҡдәәйғҪзҹҘйҒ“жӯ»й”ҒжҳҜжҖҺд№ҲдёҖеӣһдәӢе„ҝпјҡзәҝзЁӢAе’ҢзәҝзЁӢBзӣёдә’зӯүеҫ…еҜ№ж–№жҢҒжңүзҡ„й”ҒеҜјиҮҙзЁӢеәҸж— йҷҗжӯ»еҫӘзҺҜдёӢеҺ»гҖӮеҪ“з„¶д№ҹд»…йҷҗдәҺжӯӨдәҶпјҢй—®дёҖдёӢжҖҺд№ҲеҶҷдёҖдёӘжӯ»й”Ғзҡ„зЁӢеәҸе°ұдёҚзҹҘйҒ“дәҶпјҢиҝҷз§Қжғ…еҶөиҜҙзҷҪдәҶе°ұжҳҜдёҚжҮӮд»Җд№ҲжҳҜжӯ»й”ҒпјҢжҮӮдёҖдёӘзҗҶи®әе°ұе®ҢдәӢе„ҝдәҶпјҢе®һи·өдёӯзў°еҲ°жӯ»й”Ғзҡ„й—®йўҳеҹәжң¬дёҠжҳҜзңӢдёҚеҮәжқҘзҡ„гҖӮ

зңҹжӯЈзҗҶи§Јд»Җд№ҲжҳҜжӯ»й”ҒпјҢиҝҷдёӘй—®йўҳе…¶е®һдёҚйҡҫпјҢеҮ дёӘжӯҘйӘӨпјҡ

пјҲ1пјүдёӨдёӘзәҝзЁӢйҮҢйқўеҲҶеҲ«жҢҒжңүдёӨдёӘObjectеҜ№иұЎпјҡlock1е’Ңlock2гҖӮиҝҷдёӨдёӘlockдҪңдёәеҗҢжӯҘд»Јз Ғеқ—зҡ„й”Ғпјӣ

пјҲ2пјүзәҝзЁӢ1зҡ„run()ж–№жі•дёӯеҗҢжӯҘд»Јз Ғеқ—е…ҲиҺ·еҸ–lock1зҡ„еҜ№иұЎй”ҒпјҢThread.sleep(xxx)пјҢж—¶й—ҙдёҚйңҖиҰҒеӨӘеӨҡпјҢ50жҜ«з§’е·®дёҚеӨҡдәҶпјҢз„¶еҗҺжҺҘзқҖиҺ·еҸ–lock2зҡ„еҜ№иұЎй”ҒгҖӮиҝҷд№ҲеҒҡдё»иҰҒжҳҜдёәдәҶйҳІжӯўзәҝзЁӢ1еҗҜеҠЁдёҖдёӢеӯҗе°ұиҝһз»ӯиҺ·еҫ—дәҶlock1е’Ңlock2дёӨдёӘеҜ№иұЎзҡ„еҜ№иұЎй”Ғ

пјҲ3пјүзәҝзЁӢ2зҡ„run)(ж–№жі•дёӯеҗҢжӯҘд»Јз Ғеқ—е…ҲиҺ·еҸ–lock2зҡ„еҜ№иұЎй”ҒпјҢжҺҘзқҖиҺ·еҸ–lock1зҡ„еҜ№иұЎй”ҒпјҢеҪ“з„¶иҝҷж—¶lock1зҡ„еҜ№иұЎй”Ғе·Із»Ҹиў«зәҝзЁӢ1й”ҒжҢҒжңүпјҢзәҝзЁӢ2иӮҜе®ҡжҳҜиҰҒзӯүеҫ…зәҝзЁӢ1йҮҠж”ҫlock1зҡ„еҜ№иұЎй”Ғзҡ„

иҝҷж ·пјҢзәҝзЁӢ1вҖізқЎи§үвҖқзқЎе®ҢпјҢзәҝзЁӢ2е·Із»ҸиҺ·еҸ–дәҶlock2зҡ„еҜ№иұЎй”ҒдәҶпјҢзәҝзЁӢ1жӯӨж—¶е°қиҜ•иҺ·еҸ–lock2зҡ„еҜ№иұЎй”ҒпјҢдҫҝиў«йҳ»еЎһпјҢжӯӨж—¶дёҖдёӘжӯ»й”Ғе°ұеҪўжҲҗдәҶгҖӮд»Јз Ғе°ұдёҚеҶҷдәҶпјҢеҚ зҡ„зҜҮе№…жңүзӮ№еӨҡгҖӮ

public class DeadLockDemo {
 
	public static void main(String[] args) {
		Object lockA = new Object();
		Object lockB = new Object();
		new Thread(new Runnable() {
			@Override
			public void run() {
				String name = Thread.currentThread().getName();
				synchronized (lockA) {
					System.out.println(name + " got lockA,  want LockB");
					try {
						Thread.sleep(100);
					} catch (InterruptedException e) {
 
						e.printStackTrace();
					}
					synchronized (lockB) {
						System.out.println(name + " got lockB");
						System.out.println(name + ": say Hello!");
					}
				}
			}
		}, "зәҝзЁӢ-A").start();
 
		new Thread(new Runnable() {
			@Override
			public void run() {
 
				String name = Thread.currentThread().getName();
				synchronized (lockB) {
					System.out.println(name + " got lockB, want LockA");

					}
					synchronized (lockA) {
						System.out.println(name + " got lockA");
						System.out.println(name + ": say Hello!");
					}
				}
 
			}
		}, "зәҝзЁӢ-B").start();
	}
}
