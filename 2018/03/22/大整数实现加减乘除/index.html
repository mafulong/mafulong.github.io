# 大整数操作
class BigInt:
    def add(self, a, b):
        i, j = len(a) - 1, len(b) - 1
        carry = 0
        res = []
        while i >= 0 or j >= 0 or carry > 0:
            if i >= 0:
                carry += int(a[i])
                i -= 1
            if j >= 0:
                carry += int(b[j])
                j -= 1
            res.append(str(carry % 10))
            carry = carry // 10
        res = "".join(res[::-1])
        return res

    def multiply(self, a, b):
        if a == "0" or b == "0":
            return "0"

        a = list(a[::-1])
        b = list(b[::-1])
        r = ['0'] * (len(a) + len(b) + 1)
        for i in range(len(a)):
            up = 0
            for j in range(len(b)):
                t = (int(a[i]) - int('0')) * (int(b[j]) - int('0')) + up + int(r[i + j]) - int('0')
                up = t // 10
                r[i + j] = chr(ord('0') + t % 10)
            if up != 0:
                r[i + j + 1] = chr(up + ord('0'))

        res = ''.join(r)
        for i in range(len(res) - 1, -1, -1):
            if res[i] != '0':
                res = res[:i + 1]
                break
        return res[::-1]

    def sub(self, a, b):
        # print("sub ", a,b)
        if a == b:
            return "0"
        isf = False  # a，b互换
        if len(a) < len(b) or (len(a) == len(b) and a < b):
            a, b = b, a
            isf = True

        a = a[::-1]
        b = b[::-1]
        r1 = list(map(int, a))
        r2 = list(map(int, b))
        for i in range(len(r2)):
            if r1[i] >= r2[i]:
                r1[i] = r1[i] - r2[i]
            else:
                r1[i + 1] = r1[i + 1] - 1
                r1[i] = r1[i] - r2[i] + 10
        r1 = list(map(str, r1))
        res = ''.join(r1[::-1])
        for i in range(len(res)):
            if res[i] != '0':
                res = res[i:]
                break
        if isf:
            res = "-" + res
        return res

    def divide(self, a, b):
        res = ""
        if (len(a) == len(b) and a < b) or len(a) < len(b):
            return "0"
        dis = len(a) - len(b)
        len2 = len(b)
        b += "0" * dis

        while dis >= 0:
            sum = 0
            while len(b) < len(a) or (len(a) == len(b) and a >= b):
                sum += 1
                a = self.sub(a, b)
            res += str(sum)
            dis -= 1
            b = b[:len2 + dis]
        # a is the remainder
        # print(" 余数:  " + a)
        for i in range(len(res)):
            if res[i] != '0':
                res = res[i:]
                break
        return res

乘法

还是模拟手算，第一个字符串的第i位乘以第二个字符串的第j位一定是结果的第i+j位，如果i+j已经有值，直接加上去就OK，别忘了处理进位。

这样的算法的复杂度是O（n2）.利用FFT可以将算法优化到O（nlogn），关于FFT的实现在此不再赘述，可以参考算法导论或者 http://www.cnblogs.com/lsx54321/archive/2012/07/20/2601632.html。

除法

我们将除法看作是减法来处理，用被减数不断的减去减数，记录减的次数即是商的值。但是我们当然不能一个一个减，因为如果有10000000/1这种情况不是要减到猴年马月。

我们可以记录被减数和减数的位数之差len，将减数扩大10的len倍。然后依次去减，一旦被减数小于减数时，将减数减小10倍，直至到原值。依次循环，去掉前置0，得出结果。

    static String sub(String a,String b){
        if(a.compareTo(b)==0){

        boolean isf=false;
        if(a.length()<b.length()||a.length()==b.length()&&a.compareTo(b)<0){

        a=new StringBuilder(a).reverse().toString();
        b=new StringBuilder(b).reverse().toString();
        char[] r1=a.toCharArray();
        char[] r2=b.toCharArray();
        for(int i=0;i<r2.length;i++){
            char t1=r1[i];

            if(t1>=t2){

    /**
     * 大整数加法
     * @param a
     * @param b
     * @return
     */
    static String add(String a,String b){
        String res="";
        int i=a.length()-1;
        int j=b.length()-1;
        int carry=0;
        while (i>=0||j>=0||carry>0){
            if(i>=0){
                carry+=a.charAt(i--)-'0';
            }
            if(j>=0){
                carry+=b.charAt(j--)-'0';
            }
            res=String.valueOf(carry%10)+res;
            carry=carry/10;
        }
        return res;
    }

    /**
     * 大整数乘法
     * @param a1
     * @param b1
     * @return
     */
    static String multiply(String a1,String b1){
        if(a1.equals("0")||b1.equals("0")){
            return "0";
        }
        a1=new StringBuilder(a1).reverse().toString();
        b1=new StringBuilder(b1).reverse().toString();
        char[] a=a1.toCharArray();
        char[] b=b1.toCharArray();
        char[] r=new char[a.length+b.length+1];
        Arrays.fill(r,'0');
        int up=0;
        for(int i=0;i<a.length;i++){
            up=0;
            int j;
            for(j=0;j<b.length;j++){
                int t=(a[i]-'0')*(b[j]-'0')+up+r[i+j]-'0';
                up=t/10;
                r[i+j]=(char)('0'+t%10);
            }
            if(up!=0){
                r[i+j]+=up+'0';
