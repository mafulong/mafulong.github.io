#include<iostream>
#include<string>
#include<cctype>
#include<vector>
#include<algorithm>
#include<functional>
using namespace std;

int c[11];
int main() {
#ifdef LOCAL
	freopen("d:\\datain.txt", "r", stdin);
	//freopen("d:\\dataout.txt", "w", stdout);
#endif LOCAL
	fill(c, c + 11, 0);
	string n;
	cin >> n;
	for (int i = 0; i < n.length(); i++) {
		c[n[i] - '0']++;
	}
	for (int i = 0; i < 10; i++) {
		if (c[i] > 0) {
			printf("%d:%d\n", i, c[i]);
		}
	}
}
