```c++
#include<iostream>
#include<string>
#include<cctype>
#include<vector>
#include<algorithm>
using namespace std;
int main() {
#ifdef LOCAL
	freopen("d:\\datain.txt", "r", stdin);
	//freopen("d:\\dataout.txt", "w", stdout);
#endif LOCAL
	string a, b;
	char da, db;
	cin >> a >> da >> b >> db;
	long long a1=0, b1=0;
	int ca = 0, cb = 0;
	for (int i = 0; i < a.size(); i++) {
		if (a[i] == da)
			ca++;
	}
	for (int i = 0; i < b.size(); i++) {
		if (b[i] == db)
			cb++;
	}
	if (ca != 0)
		a1 = da - '0';
	if (cb != 0)
		b1 = db - '0';
	for (int i = 1; i < ca; i++)
		a1 = a1 * 10 + da - '0';
	for (int i = 1; i < cb; i++)
		b1 = b1 * 10 + db - '0';

	cout << a1 + b1;
}
```
