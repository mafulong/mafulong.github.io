#include <iostream>
#include <algorithm>
#include<math.h>
#include<string>
using namespace std;
int main() {
#ifdef LOCAL
	freopen("D:\\datain.txt", "r", stdin);
#endif
	string max1, min1;
	int count = 0;
	int n;
	cin >> n;
	string t1,t2;
	string min2 = "1814/09/06";
	string max2 = "2014/09/06";
	max1 = min2;
	min1 = max2;
	string min3, max3;
	for (int i = 0; i < n; i++) {
		cin >> t1 >> t2;
		if (t2 >= min2 && t2 <= max2) {
			count++;
			if (t2 >= max1) {
				max1 = t2;
				max3 = t1;
			}
			if (t2 <= min1) {
				min1 = t2;
				min3 = t1;
			}
		}
	}
	cout << count;
	if(count>0)
		cout<<" " << min3 << " " << max3;
	return 0;
}
