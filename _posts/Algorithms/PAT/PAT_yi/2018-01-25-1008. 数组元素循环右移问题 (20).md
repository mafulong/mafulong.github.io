```c++
#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<string.h>
#include<map>
#include<string>
#include<fstream>
#include<vector>
#include<functional>
using namespace std;

int main() {
#ifdef LOCAL
	freopen("d:\\datain.txt", "r", stdin);
	freopen("d:\\dataout.txt", "w", stdout);
#endif LOCAL
	int n, m;
	int d[100];
	
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> d[i];
	}
	m = m % n;
	if (n > m) {
		for (int j = n - m; j < n; j++) {
			cout << d[j] << " ";
		}
		for (int i = 0; i < n - m - 1; i++) {
			cout << d[i] << " ";
		}
		cout << d[n - m - 1];
	}
	else {
		for (int i = 0; i < n-1; i++) {
			cout << d[i] << " ";
		}
		cout << d[n - 1];
	}
	return 0;
}
```
