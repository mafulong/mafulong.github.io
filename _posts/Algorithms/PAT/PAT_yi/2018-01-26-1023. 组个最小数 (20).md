#ifdef LOCAL

	int c[11];
	for (int i = 0; i < 10; i++) {
		scanf("%d", &c[i]);
	}
	int mind = 0;
	if (c[0] > 0) {
		for (int i = 1; i < 10; i++) {
			if (c[i] > 0) {
				mind = i;
				cout << mind;
				c[i]--;
				break;
			}
		}

	}
	for (int i = 0; i < 10; i++) {
		for (int j = 0; j < c[i]; j++) {
			printf("%d", i);
		}
	}
}
