```c++
//cpp:
#include <iostream>
#include <algorithm>
#include<math.h>
using namespace std;
int main() {
#ifdef LOCAL
	freopen("D:\\datain.txt", "r", stdin);
#endif
	int a;
	char b;
	scanf("%d %c", &a, &b);
	int n = floor(sqrt((float)(a+1)/2));
	int yushu = a - 2 * pow(n,2) + 1;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < i; j++)
			printf(" ");
		for (int j = 0; j < 2 * (n - i) - 1; j++)
			printf("%c", b);
		printf("\n");
	}
	for (int i = 1; i < n; i++) {
		for (int j = 0; j < n - i-1; j++)
			printf(" ");
		for (int j = 0; j < 2 * i + 1; j++)
			printf("%c", b);
		printf("\n");

	}
	cout << yushu;
	return 0;
}
```
