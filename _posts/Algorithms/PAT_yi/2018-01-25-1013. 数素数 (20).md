bool isprime(int d) {
	for (int i = 2; i*i <= d; i++) {
		if (d%i == 0) {
			return false;
		}
	}
	return true;
}
int main() {

	for (int i = 2;; i++) {
		if (isprime(i)) {
			j++;

				cout << i;

				else {
					cout << " ";
				}
			}
		}
	}
	return 0;
}
