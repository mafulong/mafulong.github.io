title: C++ swap()函数
---

```c++
#include<iostream>
#include<string>
#include<cstring>
#include<stdio.h>
#include<vector>
using namespace std;
int main() {
    vector<int> a;
    a.push_back(3);
    a.push_back(4);
    swap(a[0],a[1]);
    cout<<a[0]<<" "<<a[1]<<endl;
}
```
