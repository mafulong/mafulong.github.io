## 反转

string:

    string s=str;
    reverse(s.begin()+2,s.begin()+7);//和下面一句效果一样

char[]

    char str[]="abcdefgh";
    reverse(str+2,str+7);
    or
    reverse(begin(str),begin(str)+7)

vector

    reverse(begin(a), begin(a) + n);
    reverse(begin(a), begin(a) + m);
    reverse(begin(a) + m, begin(a) + n);
