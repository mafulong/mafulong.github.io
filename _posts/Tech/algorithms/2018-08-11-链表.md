## 完整代码
```java
import java.io.*;
import java.util.Arrays;
import java.util.Scanner;

import static java.lang.Math.sqrt;

public class c1 {
    static class Node{
        int data;
        Node next;
        public Node(int data){
            this.data=data;
        }

        @Override
        public String toString() {
            return "Node{" +
                    "data=" + data +
                    ", next=" + next +
                    '}';
        }
    }

    /**
     * 构造链表，尾插法
     * @param data
     * @return
     */
    static Node createList(int[] data){
        Node p1=null;
        Node head=new Node(0);
        p1=head;
        for(int i=0;i<data.length;i++){
            Node p=new Node(data[i]);
            p1.next=p;
            p1=p1.next;
        }
        return head.next;
    }

    /**
     * 反转链表
     * @param head
     * @return
     */
    static Node reverseList(Node head){
        Node newhead=null;
        Node pre=null,pnext=null;
         while(head!=null){
//            System.out.println(head.data);
            pnext=head.next;
            head.next=pre;
            pre=head;
            newhead=pre;
            head=pnext;

        }
        return newhead;
    }
    //全新精简版本，要啥新的头节点啊
    static Node newReverse(Node head){
        Node pre=null;
        while (head!=null){
            Node t=head.next;
            head.next=pre;
            pre=head;
            head=t;
        }
        return pre;
    }
    /**
     * 删除节点，可以建立个头节点
     * @param head
     * @param key
     * @return
     */
    static Node deleteNode(Node head,int key){
        Node newhead=new Node(0);
        newhead.next=head;
        Node pre=newhead;
        while(head!=null){
            if(head.data==key){
                pre.next=head.next;
                head=head.next;
                continue;
            }
            pre=head;
            head=head.next;
        }

        return newhead.next;
    }

    /**
     * 插入节点
     * @param head
     * @param index
     * @param key
     * @return
     */
    static Node insertNode(Node head,int index,int key){
        Node pre=new Node(0);
        pre.next=head;
        Node newhead=pre;
        while(head!=null&&index-->0){
            pre=head;
            head=head.next;
        }
        Node p=new Node(key);
        pre.next=p;
        p.next=head;
        return newhead.next;
    }
    public static void main(String[] args){
        int[] d=new int[]{1,3,3,2,23,3,4,1,2};
        Node head=createList(d);
//        head = reverseList(head);
        insertNode(head,1,2222);
        while(head!=null){
            System.out.println(head.data);
            head=head.next;
        }

    }
}
```
