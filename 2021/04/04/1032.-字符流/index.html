class StreamChecker:

    def __init__(self, words: List[str]):
        self.trie = {}
        for word in words:
            node = self.trie
            for c in reversed(word):
                if c not in node:
                    node[c] = {}
                node = node[c]
            node['#'] = None
        self.history = []

    def query(self, letter: str) -> bool:
        self.history.append(letter)
        node = self.trie
        for i in range(len(self.history) - 1, -1, -1):
            c = self.history[i]
            if c not in node:
                return False
            node = node[c]
            if '#' in node:
                return True
        return False
