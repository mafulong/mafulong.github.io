class Solution:
    def countNodes(self, root: TreeNode) -> int:
        if not root:
            return 0
        height = 0
        p = root
        while p:
            height += 1
            p = p.left
        # print(height)
        left, right = (1 << (height - 1)), (1 << height) - 1
        # print(left, right)
        while left <= right:
            mid = (right - left) // 2 + left
            if self.exist(root, mid, height):
                left = mid + 1
            else:
                right = mid - 1
        return right

    def exist(self, root: TreeNode, k: int, height: int):
        if height == 1:
            return root is not None
        bits = 1 << (height - 2)
        while root and bits > 0:
            if bits & k == 0:
                root = root.left
            else:
                root = root.right
            bits >>= 1
        return root is not None
