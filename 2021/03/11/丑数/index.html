使用优先队列

class Solution:
    def nthSuperUglyNumber(self, n: int, primes: List[int]) -> int:
        import heapq
        hq = [1]
        res = []
        for _ in range(n):
            top = heapq.heappop(hq)
            res.append(top)
            while hq and hq[0] == top:
                heapq.heappop(hq)
            for k in primes:
                heapq.heappush(hq, top * k)
        print(res)
        return res[-1]
