class Solution:
    def minTime(self, time: List[int], m: int) -> int:
        n = len(time)

        def check(k):
            cnt, sumv = 0, 0
            maxTime = 0
            for i in range(n):
                v = time[i]
                maxTime = max(maxTime, v)
                if sumv + v - maxTime > k:
                    sumv = v
                    cnt += 1
                    maxTime = v
                else:
                    sumv += v
            if sumv:
                cnt += 1
            return cnt <= m

        l, r = 0, sum(time)
        while l <= r:
            mid = (l + r) // 2
            if check(mid):
                r = mid - 1
            else:
                l = mid + 1
        return l
