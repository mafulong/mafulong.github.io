class Solution:

    def __init__(self, w: List[int]):
        self.w = w
        total = 0
        self.w2id = {}
        for i, v in enumerate(w):
            total += v
            self.w2id[total] = i
        self.total = total
        self.keys = list(self.w2id.keys())

    def pickIndex(self) -> int:
        import random
        import bisect
        rand = random.randint(1, self.total)
        id = bisect.bisect_left(self.keys, rand)
        return self.w2id[self.keys[id]]
