维护map

560. 和为K的子数组

记录前缀和，利用hash表存储sum 到 count的数量，遍历时累加。


class Solution {
public:
	int subarraySum(vector<int>& nums, int k) {
		int sum = 0, res = 0;
		unordered_map<int, int> map1;
		map1[0] = 1;
		for (int i = 0; i < nums.size(); i++) {
			sum += nums[i];
			if (map1.count(sum - k)) {
				res += map1.at(sum - k);
			}
			map1[sum]++;
		}
		return res;
	}
};
