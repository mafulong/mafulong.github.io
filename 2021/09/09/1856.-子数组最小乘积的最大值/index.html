class Solution:
    def maxSumMinProduct(self, nums: List[int]) -> int:
        n = len(nums)
        mod = 10 ** 9 + 7
        L, R = [-1] * n, [n] * n
        stack = []
        pre = [0] * (n + 1)
        for i, v in enumerate(nums):
            pre[i + 1] = pre[i] + v
            while stack and nums[stack[-1]] > v:
                R[stack[-1]] = i
                stack.pop()
            if stack:
                L[i] = stack[-1]
            stack.append(i)
        ans = 0
        for i, v in enumerate(nums):
            l,r = L[i]+1, R[i]-1
            ans = max(ans, (pre[r + 1] - pre[l]) * v)
        return ans % mod
